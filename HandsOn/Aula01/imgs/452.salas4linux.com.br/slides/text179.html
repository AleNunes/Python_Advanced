    • 1# ls /var/lib/ldap/

    2# /etc/init.d/slapd stop

    3# slapcat -l /etc/ldap/backup.base.ldap

    4# rm -rf /var/lib/ldap/*

    5# /etc/init.d/slapd start

    6# ldapsearch -x -LLL

    7# /etc/init.d/slapd stop

    8# slapadd -l /etc/ldap/backup.base.ldap

    9# /etc/init.d/slapd start
