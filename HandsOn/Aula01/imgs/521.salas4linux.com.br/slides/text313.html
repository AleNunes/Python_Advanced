def getProjectRepo(self,project):

response = self.make_get(“projects”,”/all”)

project = min([ r.get(“ssh_url_to_repo”) for r in response.json() if r.get(“name”) == project ])

if project:

return str(project)

else:

print “Projeto não existe”
