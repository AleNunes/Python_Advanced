def runCommand(self,command):

stdin,stdout,stderr = self.client.exec_command(command)

if stderr.channel.recv_exit_status() != 0:

return {“status”:1,”message”:stderr.read()}

else:

Return {“status”:0,”message”:stdout.read()}
