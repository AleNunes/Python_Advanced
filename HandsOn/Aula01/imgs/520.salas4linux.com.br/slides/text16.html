Strings

  • A concatenação de string também pode ser feita através de parâmetros similares à linguagem C.

string = “ %s é uma linguagem muito legal”%curso

# é possível utilizar mais de um parâmetro

string = “Estudo %s na %s”%(curso,empresa)

curso = “Python”

empresa = “4Linux”
